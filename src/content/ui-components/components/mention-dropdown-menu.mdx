## Install

```bash
npx tiptap-cli add mention-dropdown-menu
```

## Usage

```tsx
import { MentionDropdownMenu } from "@/components/tiptap-ui/mention-dropdown-menu"
import { useEditor, EditorContent } from "@tiptap/react"
import { Mention } from "@tiptap/extension-mention"

export default function MyEditor() {
  const editor = useEditor({
    extensions: [
      // other extensions
      Mention.configure({
        // mention configuration
      }),
    ],
  })

  return (
    <div>
      <EditorContent editor={editor} />
      <MentionDropdownMenu editor={editor} />
    </div>
  )
}
```

## Props

| Name            | Type           | Default                        | Description                |
| --------------- | -------------- | ------------------------------ | -------------------------- |
| editor          | Editor \| null | null                           | The TipTap editor instance |
| floatingOptions | object         | -                              | Options for floating UI    |
| selector        | string         | 'tiptap-mention-dropdown-menu' | Selector for the target    |
| char            | string         | '@'                            | Mention character          |
