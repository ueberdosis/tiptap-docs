A popover for selecting text highlight colors.

## Install

```bash
npx tiptap-cli add highlight-popover
```

## Usage

```tsx
import { HighlightPopover } from "@/components/tiptap-ui/highlight-popover"
import { useEditor, EditorContent } from "@tiptap/react"
import { Highlight } from "@tiptap/extension-highlight"

export default function MyEditor() {
  const editor = useEditor({
    extensions: [
      // other extensions
      Highlight.configure({
        multicolor: true,
      }),
    ],
  })

  return (
    <div>
      <div className="toolbar">
        <HighlightPopover editor={editor} />
      </div>
      <EditorContent editor={editor} />
    </div>
  )
}
```

## Props

| Name                | Type             | Default                  | Description                                  |
| ------------------- | ---------------- | ------------------------ | -------------------------------------------- |
| editor              | Editor \| null   | null                     | The TipTap editor instance                   |
| colors              | HighlightColor[] | DEFAULT_HIGHLIGHT_COLORS | Array of highlight color options             |
| hideWhenUnavailable | boolean          | false                    | Whether the popover is in a floating context |
